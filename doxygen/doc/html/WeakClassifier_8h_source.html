171  virtual double error_rate (const std::list<Inputs *> &dataset, const std::list<double> &weights)
172  {
173  double e = 0.0;
174  std::list<Inputs*> ::const_iterator d_itr = dataset.begin();
175  std::list<double> ::const_iterator w_itr = weights.begin();
176  while(d_itr != dataset.begin())
177  {
178  if(classify(*d_itr) != (*d_itr)->getLabel())
179  e += *w_itr;
180  d_itr++;
181  }
182  return e;
183  }
