184 void OnlineBoost::fromString(const std::string &str)
185 {

187 
188  yarp::os::Bottle bLoader;
189  bLoader.fromString(str.c_str());
190

203  isReady();

218 void OnlineBoost::update(const Inputs *input, double &weight)
219 {
220  //if the algorithm uses a function space, extract new weak learners from the input (if it is a positive sample).
221  if(function_space.size()<max_function_space_size && input->getLabel() == +1)
222  {
223  ClassifierFactory::instance().pack(input,function_space);
224 
225  //avoid getting a function space too large
226  while(function_space.size() > max_function_space_size)
227  {
228  delete function_space.front();
229  function_space.pop_front();
230  }
231  }
232 
233  if(isReady())
234  {
235  //propagate the sample through the list of weak classifiers
236  for(unsigned int i = 0; i < weak_classifiers.size(); i++)
237  {
238  weak_classifiers[i]->update(input,weight);
239 
240  //if the WeakClassifiers abstains from classifing, do not update the weight
241  if(weak_classifiers[i]->classify(input) != 0)
242  {
243  if(weak_classifiers[i]->classify(input) == input->getLabel())
244  weight *= 0.5/(1.0-weak_classifiers[i]->getError());
245  else
246  weight *= 0.5/weak_classifiers[i]->getError();
247  }
248 
249  //update alpha
250  alphas[i] = 0.5*log((1-weak_classifiers[i]->getError())/weak_classifiers[i]->getError());
251 
252  }
253  }
254 }
