84 void AdaBoost::train(const std::list<Inputs*> &training_set)
85 {
86  //check if the classifier is ready
87  if(!isReady())
88  {
89  std::cout << "AdaBoost not yet initialized" << std::endl;
90  return;
91  }
92 
93  if(alphas.size())
94  std::cout << "AdaBoost already trained. Training will continue." << std::endl;
95 
96  //initialize the weights
97  std::list<double> weights(training_set.size(),(double) 1.0/training_set.size());
98 
99  for(int t = 0; t < max_iter; t++)
100  {
101  //find and clone the current best weak learner
102  WeakClassifier *best_w = best_classifier(training_set,weights);
103  if(best_w->getError() > 0.5)
104  {
105  std::cout << "There are not weak learners better than random guess." << std::endl;
106  std::cout << "Training is stopped." << std::endl;
107  break;
108  }
109  //add the weak learner to the strong classifier with corresponding alpha
110  weak_classifiers.push_back(best_w);
111  if(best_w->getError() < 1e-7)
112  {
113  std::cout << "The Weak Learner at iteration n. " << t << " solves the entire problem." << std::endl;
114  alphas.push_back(1.0);
115  break;
116  }
117  else
118  alphas.push_back(0.5*log((1-best_w->getError())/best_w->getError()));
119 
120 
121  //update the weights
122  double norm = 0.0;
123  std::list<Inputs*>::const_iterator t_itr = training_set.begin();
124  std::list<double>::iterator w_itr = weights.begin();
125  while(t_itr != training_set.end() && w_itr != weights.end())
126  {
127  int d = best_w->classify(*t_itr)*(*t_itr)->getLabel();
128  *w_itr *= exp(-alphas[t]*d);
129  norm += *w_itr;
130  t_itr++;
131  w_itr++;
132  }
133 
134  w_itr = weights.begin();
135  while(w_itr != weights.end())
136  {
137  *w_itr /= norm;
138  w_itr++;
139  }
140 
141  }
142 
143 }

150 WeakClassifier *AdaBoost::best_classifier(const std::list<Inputs *> &training_set, const std::list<double> &weights)
151 {
152  double best_error=1.0;
153  WeakClassifier *best_wl;
154 
155  //find the weak classifier with lowest error rate
156  for(unsigned int i = 0; i < function_space.size(); i++)
157  {
158  double error = function_space[i]->error_rate(training_set,weights);
159  if(error < best_error)
160  {
161  best_error = error;
162  best_wl = function_space[i];
163  }
164  }
165 
166  //clone the weak classifier
167  WeakClassifier *w = best_wl->create();
168  //check if the (copied) best_wl is ready
169  if(!w->isReady())
170  {
171  std::ostringstream buffer;
172  buffer << "The chosen weak learner is not ready yet";
173  throw std::runtime_error(buffer.str());
174  }
175 
176  return w;
177 }
